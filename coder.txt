#include<iostream>
#include<cstring>
#include<stack>
using namespace std;
//balanced parenthesis problem solution using stack ds
bool checkexp(string s){
	stack <char> st;
	int i;
	for(i=0;i<s.size();i++){
		char ch=s[i];
		if(ch=='('||ch=='{'||ch=='[') {
		st.push(ch);}
		
		if(ch==')')
        {if(st.empty()||st.top()!='(')
	     return false;st.pop();	}
	     if(ch==']')
        {if(st.empty()||st.top()!='[')
	     return false;st.pop();	}
	     if(ch=='}')
        {if(st.empty()||st.top()!='{')
	     return false;st.pop();	}

	     
		}	
		return st.empty();
	}

int main(){



	string s;
	cin>>s;
	bool isvalid=checkexp(s);
	if(isvalid){cout<<"YES"<<endl;
	}
	else cout<<"NO"<<endl;
	
}
